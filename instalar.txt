no arquivo prisma/schema.prisma:
datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id        Int     @default(autoincrement()) @id
  email     String  @unique
  name      String
  telefone  String?

  @@map("users")
}

model Cliente {
    id          String  @id @default(uuid()) 
    nome        String
    telefone    String?
    email       String @unique
    observacoes String

    @@map("clientes")
}

generator client {
  provider = "prisma-client-js"
}
